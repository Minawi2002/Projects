23 /* WARNING: can only be called with vectors containing elements of type T that derives from "serializable" */
24 static rapidjson::Value serialize_vector(const std::vector<T*>& serializables, rapidjson::Document::AllocatorType& allocator) {
25 derived_from<T,serializable>(); // ensure T derives from serializable
26 rapidjson::Value arr_val(rapidjson::kArrayType);
27 for (int i = 0; i < serializables.size(); i++) {
28 rapidjson::Value elem(rapidjson::kObjectType);
29 serializables[i]->write_into_json(elem, allocator);
30 arr_val.PushBack(elem, allocator);
31 }
32 return arr_val;
